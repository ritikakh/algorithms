  var j = arr.length - 1;
	while(i < j) {

  	var high = arr[j];
    if(low + high === sum) {

      return true;
    }
    else if(low + high > sum) {
    	j--;
      //continue;
    }

    }
  }
}

checkPair(arr, sum);
